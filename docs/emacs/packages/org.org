Org is a powerful and extensible framework for text processing, driven by workflows. It's used to write anything: notes, todos, literate programs, and more!
#+begin_src emacs-lisp
  (use-package org
    :straight t
    :defer t
    :hook ((org-mode . visual-line-mode)
           (org-mode . org-num-mode))
    :init
    (auto-fill-mode -1)
    :config
    (require 'org-protocol)
    (setq org-directory "~/org"
          org-default-notes-file (expand-file-name "notes.org" org-directory))
    <<org-capture>>
   )
#+end_src

        '(
          ("g" "General To-Do"
